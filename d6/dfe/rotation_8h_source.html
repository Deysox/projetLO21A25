12 /*
13 * Représente la rotation d'une case hexagonale ou d'une tuile dans le jeu Akropolis.
14 * La rotation peut prendre six valeurs possibles, modélisé par un entier r allant de 0 à 5.
15 * L'objet Rotation correspond à r Rotations successives de 60° dans le sens horaire
16 * à partir de la position initiale.
17 *
18 * Pour les cases individuelles non bicolores, la rotation n'a aucun impact.
19 * Pour les cases individuelles bicolores, la rotation influe sur les relations de connexion entre couleurs sur
20 * le plateau. La moitié des voisins seront considérés comme adjacents à une tuile monochrome de la première couleur,
21 * tandis que l'autre moitié le sera avec la seconde.
22 *
23 * Pour les tuiles à trois cases, la rotation se présente comme suit.
24 * /A_\
25 * \__//B_\
26 * /PP\\__/
27 * \PP/
28 * La rotation s'effectue dans le sens horaire, autour de la tuile pivot marquée P. Une Rotation 1 sur cette tuile
29 * donnerait alors :
30 *
31 * /A_\
32 * /PP\\__/
33 * \PP//B_\
34 * \__/
35 */
36 class Rotation {
37 int r;
38 public:
39 /*
40 * Constructeur de la classe Rotation.
41 * @param rot Valeur initiale de la rotation
42 */
43 Rotation(int rot = 0) : r(rot%6) {};
44
45 /*
46 * Accesseur en lecture de l'attribut r
47 * @return entier correspondant à la rotation, entre 1 et 6
48 */
49 int value() const {return r;}
50
51 /*
52 * Additionne deux rotations
53 * @param Second opérande
54 * @return Objet Rotation correspondant à la rotation totale effectuée par cumul des rotations
55 * entraînées par les opérandes.
56 */
57 Rotation operator+(const Rotation& other) const {return {(r+other.r)%6};};
58
59 /*
60 * Soustrait deux rotations
61 * @param Second opérande
62 * @return Objet Rotation correspondant à la rotation totale effectuée par cumul des rotations
63 * entraînées dans le sens horaire pour le premièr opérande
64 * et dans le sens inverse pour le second opérande.
65 */
66 Rotation operator-(const Rotation&) const {return {(r-1)%6};}
67
68 /*
69 * Incrémente la rotation d'un sixième de tour
70 * @return Rotation modifiée
71 */
72 Rotation& operator++() {r++; return *this;}
73 /*
74 * Incrémente la rotation d'un sixième de tour
75 * @return Rotation non modifiée
76 */
77 Rotation operator++(int) {int old = r; r++; return {old};};
78 /*
79 * Compare deux rotations
80 * @param other Second opérande
81 * @return true si les deux rotations ont la même valeur
82 */
83 bool operator==(const Rotation& other) const {return r == other.r;};
84 /*
85 * Compare deux rotations
86 * @param other Second opérande
87 * @return false si les deux rotations ont la même valeur
88 */
89 bool operator!=(const Rotation& other) const {return r != other.r;}
90 };
91}
